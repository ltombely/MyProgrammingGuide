Constraints

  • NOT NULL
  • DEFAULT '---'
  • AUTO_INCREMENT > Primeiro a cadastras será o 1, segundo o 2...
  • PRIMARY KEY (__) > parâmetro recebe o nome do campo criado. Primary key impede que tenha outro cadastro com a mesma informação
  • UNIQUE
  • UNSIGNED > Faz com que o valor seja obrigatoriamente positivo

Comandos DDL MySQL

COMANDO Função Caraceterística
CREATE DATABASE ---; Cria um banco de dados O Parâmetro recebe o nome do BD que se quer criar
USE ---; Seleciona qual banco vai receber o comando Usar esse comando ou clicar no banco para selecioná-lo
CREATE TABLE ---( --- ); Cria uma Tabela Primeiro Parâmetro recebe o nome da tabela. Segundo recebe as características dos registros

DROP DATABASE ---; Deleta banco de nome ---
DEFAULT CHARACTER SET utf8 Faz com que o tipo de letra do DB seja UTF-8 Esse comando se coloca ao criar o DB após o nome do DB
DEFAULT COLLATE utf8_general_ci Define os caracteres, o collatio do DB Usa-se ao criar o DB após o nome dele

Command Examples:

                    
Create Database:

            
            CREATE DATABASE cadastro
            DEFAULT CHARACTER SET utf8
            DEFAULT COLLATE utf8_general_ci;
                    
Create Table:

            
            CREATE TABLE pessoas(
                id int NOT NULL AUTO_INCREMENT,
                nome varchar(30) NOT NULL,
                nascimento date,
                sexo enum('M','F'),
                peso decimal(5,2),
                altura decimal(3,2),
                nacionalidade varchar(20) DEFAULT 'Brasil',
                PRIMARY KEY (id)
            )DEFAULT CHARSET = utf8;

Comandos DML

COMANDO Função Características
INSERT INTO tabela1 (colunas_da_tabela)2 VALUES (valores)3; Insere na tabela pessoas com características (__)2 os valores de (__)3 ___1 Recebe o nome da tabela. Caso o dado esteja já predefinido na criação da tabela, não é necessário colocar o dado.
INSERT INTO tabela1 VALUES (valores)2; Inserir na tabela ___1 os dados (__)2 Caso os dados estejam completos, pode-se abreviar dessa forma.
SELECT _1 FROM ___2 Seleciona -1 da tabela ___2 Se colocar * no _1, seleciona todos
ALTER TABLE ___ Diz qu vai alterar uma tabela Passar o comando a se fazer em seguida
ADD COLUMN nome_coluna (tipo_primitivo) Adiciona um novo campo no banco Necessário digitar ALTER TABLE __ antes. Column significa campo em MySQL
DROP COLUMN nome_coluna Deleta coluna Necessário usar ALTER TABLE __ antes
MODIFY COLUMN nome_coluna tipo_primitivo Modifica tipo primitivo ou constraints Usa-se após ALTER TABLE __. Column pode ser omitido
CHANGE COLUMN nome_antigo nome_novo tipo_primitivo constraints Altera o nome de uma tabela Usa-se com ALTER TABLE
RENAME TO novo_nome; Renomea o nome de uma tabela Usa-se com ALTER TABLE
AFTER nome_coluna_anterior Inserir uma nova coluna em um lugar específico Para fazer com que seja a primeira coluna, usa-se se FIRST ao invés de AFTER
CREATE TABLE IF NOT EXISTS Se não existir, cria tabela IF NOT EXIST ou IF EXISTS são complementos para CREATE
